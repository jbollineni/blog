~~~ bash
docker run -d \
  --net=host \
  --name=mosquitto \
  -v source="/opt/mosquitto/config",destination=/mosquitto/config/ \
  -v source="/opt/mosquitto/data",destination=/mosquitto/data/ \
  -v source="/var/log/mosquitto/",destination=/mosquitto/log/ \
  eclipse-mosquitto:latest
~~~
